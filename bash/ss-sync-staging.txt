## SS-SYNC DUPLICATES ALL RELEVANT FILES FROM PRODUCTION TO STAGING (BUT NOT UPLOADS) ##
## STAGING SITES WILL MIRROR (NGINX ALIAS) ALL MEDIA UPLOADS FROM PRODUCTION ##

## source ss-config ##
source /var/www/ss-config

## source ss-functions ##
source /var/www/ss-functions

## BELOW THIS RELIES ON SS-CONFIG AND SS-FUNCTIONS

####################################################################################################
#### TABLE OF CONTENTS (SS-Sync-Staging) ###########################################################
####################################################################################################

## this is a brief summary of the different code snippets you will find in this script ##
## each section should be commented so you understand what is being accomplished ##

## A. Query Settings
## A. Touch Timestamp File
## B. Message (Begin Script)
## C. Import Production Database To Staging Database
## D. Copy All Relevant Files From Production To Staging
## E. Fix Any Hardcoded Links In Staging Theme Files


### this script should probably also run ss-install-wp-config and ss-install-mu-plugins 
## if not yet exists under staging folder and ss-install-mysql-database if not yet exists "staging"

####################################################################################################
#### A. SS-Sync-Staging: Touch Timestamp File ######################################################
####################################################################################################

## this is a dummy timestamp file that will remember the last time this script was run ##
## it can be useful for developer reference and is sometimes used by SlickStack ##

## script timestamp ##
ss_touch "${TIMESTAMP_SS_SYNC_STAGING}"

####################################################################################################
#### A. Query Settings
####################################################################################################

## check if SlickStack dashboard staging.wp_options has paused this script before proceeding

# select EmployeeName from DemoTable where isMarried=true;
# ss_options_staging_sync_status=$(mysql -BNe "USE staging; SELECT EmployeeName FROM DemoTable WHERE id=1")
# $sql = "SELECT 'id' FROM Users WHERE username='$name'";

# if [[ $ss_options_staging_sync_status == "false" ]]; then
#    echo -e "${COLOR_WARN}Exiting ss-sync-staging: Staging site sync is disabled in the SS dashboard so exiting now... ${NOCOLOR}"
#    exit 1
# fi

if [[ "${SS_SYNC_STAGING}" == "false" ]]; then
    echo -e "${COLOR_INFO}Exiting ss-sync-staging: Staging sync is disabled in ss-config so exiting the script now... ${NOCOLOR}"
    exit 1
fi

####################################################################################################
#### B. SS-Sync-Staging: Message (Begin Script) ####################################################
####################################################################################################

## this is a simple message that announces to the shell the purpose of this bash script ##
## it will only be noticed by sudo users who manually call ss core bash scripts ##

## echo message ##
echo -e "${COLOR_INFO}Running ss-sync-staging: Syncs relevant files and database from production site to the staging site... ${NOCOLOR}"
sleep "${SLEEP_MESSAGE_BEGIN}"

####################################################################################################
#### SS-Sync-Staging: Run SS-Dump-Database (Conditional) ###########################################
####################################################################################################

## we must run ss-dump-database now if no database dump file is found under backups ##
## this ensures that a dump file exists before atempting to sync it to staging ##

## FIX: change this to be OUTDATED instead e.g. more than X hours old ##
if [[ ! -f "/var/www/backups/mysql/production.sql" ]]; then
    source "${PATH_SS_DUMP_DATABASE}"
fi

####################################################################################################
#### C. SS-Sync-Staging: Import Production Database To Staging Database ############################
####################################################################################################

## this snippet duplicates the dumped production database and imports it into MySQL ##
## then it forecfully replaces the home/siteurl options to staging subdomain ##

## only if staging site is enabled ##
if [[ "${STAGING_SITE}" == "true" ]]; then

    ## import database ##
    cp /var/www/backups/mysql/production.sql /tmp/staging.sql
    chown mysql:mysql /tmp/staging.sql ## does not matter if -L flag used below
    chmod 0644 /tmp/staging.sql ## does not matter if -L flag used below
    ss_mysql_8.0 staging < /tmp/staging.sql
    
    ## fix staging home and siteurl fields (no mass search/replace to avoid potential conflicts) ##
    if [[ "${SS_APP}" == "wordpress" ]]; then
        if [[ -n $(mysql --execute "SHOW TABLES FROM staging WHERE Tables_in_staging LIKE '${DB_PREFIX}options';" 2> /dev/null) ]]; then
            mysql --execute="UPDATE staging.${DB_PREFIX}options SET option_value='https://staging.${SITE_DOMAIN_EXCLUDING_WWW}' WHERE option_name='home'";
            mysql --execute="UPDATE staging.${DB_PREFIX}options SET option_value='https://staging.${SITE_DOMAIN_EXCLUDING_WWW}' WHERE option_name='siteurl'";
        fi
    fi
    
####################################################################################################
#### D. SS-Sync-Staging: Copy All Relevant Files From Production To Staging ########################
####################################################################################################

## here we copy over everything from production to staging besides uploads (media files) ##
## schedule ss-clean beforehand in order to remove various (large) junk files ##

    ## copy wp core directories ##
    ss_rsync /var/www/html/wp-admin/ /var/www/html/staging/wp-admin
    ss_rsync /var/www/html/wp-includes/ /var/www/html/staging/wp-includes

    ## copy wp core root php files (only) ##
    # --exclude-from="/var/www/meta/exclude.txt" 
    ss_rsync --exclude="wp-config*.php*" --include="index.php" --include="wp-*.php" --exclude="*" /var/www/html/ /var/www/html/staging

    ## copy wp-content files over (skips media uploads) ##
    ss_rsync --max-size=5mb --exclude="mu-plugins*" --exclude="uploads*" /var/www/html/wp-content/ /var/www/html/staging/wp-content
    
    ## remove potential run-on subdirectories ##
    ss_rm /var/www/html/staging/staging*

####################################################################################################
#### E. SS-Sync-Staging: Fix Any Hardcoded Links In Staging Theme Files ############################
####################################################################################################

## many themes (or custom-coded themes) often contain hardcoded HTML assets and links ##
## here we forcefully fix these links to ensure staging sites render properly ##

    ## fix hardcoded theme links ##
    find /var/www/html/staging/wp-content/themes/ -type f -exec sed -i "s#https://${SITE_DOMAIN}#https://staging.${SITE_DOMAIN_EXCLUDING_WWW}#g" {} +
    find /var/www/html/staging/wp-content/themes/ -type f -exec sed -i "s#http://${SITE_DOMAIN}#https://staging.${SITE_DOMAIN_EXCLUDING_WWW}#g" {} +

## end parent if ##
fi

####################################################################################################
## delete tmp files 
####################################################################################################

## delete tmp files ##
rm /tmp/staging.sql
